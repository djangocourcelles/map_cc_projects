**D. Chargement des données et fusion localStorage (pitfall #3 : AVANT D3) :**
```javascript
fetch('/api/projects')
  .then(r => {
    if (!r.ok) throw new Error(`HTTP ${r.status}`);
    return r.json();
  })
  .then(projets => {
    if (projets.length === 0) {
      document.getElementById('etat-vide').style.display = 'block';
      return;
    }

    // Fusion positions sauvegardées — AVANT de passer à D3 (pitfall #3)
    const positionsSauvees = JSON.parse(localStorage.getItem(CLE_LS) || '{}');
    const n = projets.length;

    projets.forEach((p, i) => {
      if (positionsSauvees[p.id]) {
        // Positions restaurées depuis localStorage
        p.x = positionsSauvees[p.id].x;
        p.y = positionsSauvees[p.id].y;
      } else {
        // Layout initial en cercle centré (D-07)
        const angle = (2 * Math.PI * i) / n;
        p.x = LARGEUR / 2 + RAYON_CERCLE_INITIAL * Math.cos(angle);
        p.y = HAUTEUR / 2 + RAYON_CERCLE_INITIAL * Math.sin(angle);
      }
    });

    lancerSimulation(projets);
  })
  .catch(err => console.error('Impossible de charger les projets :', err.message));
```

**E. Fonctions utilitaires :**
```javascript
function sauvegarderPositions(nodes) {
  const positions = {};
  nodes.forEach(n => { positions[n.id] = { x: n.x, y: n.y }; });
  localStorage.setItem(CLE_LS, JSON.stringify(positions));
}

function couleurStroke(d) {
  if (!d.has_git) return '#6e7681';
  if (d.jours_depuis_commit === null || d.jours_depuis_commit > 30) return '#30363d';
  if (d.jours_depuis_commit <= 7) return '#58a6ff';
  return '#484f58';
}

function largeurStroke(d) {
  if (!d.has_git) return 1;
  if (d.jours_depuis_commit === null || d.jours_depuis_commit > 30) return 1;
  if (d.jours_depuis_commit <= 7) return 2;
  return 1.5;
}

function labelStack(d) {
  // Emojis ou texte fallback (D-03)
  if (!d.stacks || d.stacks.length === 0) return '';
  const avecEmoji = d.stacks.filter(s => s.emoji);
  if (avecEmoji.length > 0) {
    // Séparateur : espace fine Unicode U+2009 (UI-SPEC multi-stack)
    return avecEmoji.map(s => s.emoji).join(' ');
  }
  // Fallback D-03 : nom du dossier comme label stack — déjà dans stacks[0].stack
  return '';
}
```

**F. Fonction principale lancerSimulation(nodes) :**

Contient la création des nœuds SVG et les behaviors D3 :

1. Simulation :
```javascript
const simulation = d3.forceSimulation(nodes)
  .force('center', d3.forceCenter(LARGEUR / 2, HAUTEUR / 2))
  .force('repulsion', d3.forceManyBody().strength(-120))
  .force('collision', d3.forceCollide(d => d.rayon + 8))
  .alphaDecay(0.05)
  .on('tick', mettreAJour)
  .on('end', () => {
    // Pitfall #4 : arrêter après stabilisation
    simulation.alphaTarget(0).stop();
    sauvegarderPositions(nodes);
  });
```

2. Rendu nœuds avec key function (pitfall #5) :
```javascript
const noeuds = viewport.selectAll('g.noeud')
  .data(nodes, d => d.id)
  .join(
    enter => {
      const g = enter.append('g').attr('class', 'noeud');

      // Cercle principal
      g.append('circle')
        .attr('r', d => d.rayon)
        .attr('fill', '#161b22')
        .attr('stroke', d => couleurStroke(d))
        .attr('stroke-width', d => largeurStroke(d))
        .attr('stroke-dasharray', d => d.has_git ? null : '4 2')
        .attr('opacity', d => d.opacite);

      // Emoji(s) de stack (au-dessus du nom)
      g.append('text')
        .attr('class', 'emoji-stack')
        .attr('text-anchor', 'middle')
        .attr('dy', '-14px')
        .attr('font-size', '14px')
        .attr('fill', '#c9d1d9')
        .attr('opacity', d => d.opacite)
        .text(d => labelStack(d));

      // Nom du projet (D-09 : pas de troncature)
      g.append('text')
        .attr('class', 'nom-projet')
        .attr('text-anchor', 'middle')
        .attr('dy', '4px')
        .attr('font-size', '13px')
        .attr('font-weight', '600')
        .attr('fill', '#c9d1d9')
        .attr('opacity', d => d.opacite)
        .text(d => d.nom);

      // Meta : branche + date relative
      g.append('text')
        .attr('class', 'meta-projet')
        .attr('text-anchor', 'middle')
        .attr('dy', '18px')
        .attr('font-size', '11px')
        .attr('font-family', 'monospace')
        .attr('fill', '#8b949e')

      return g;
    },
    update => update,
    exit => exit.remove()  // pitfall #5
```
